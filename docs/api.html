Generate Access Token #

POST /clientapi/generate-token

Exchange your cxClientId and cxClientSecret for a short-lived Bearer token (1 hour). Any previously active token for the same client is automatically revoked.

Headers

Header Required Description
x-cx-client-id Yes Your Client Exchange ID
x-cx-client-secret Yes Your Client Exchange Secret

Code Examples

cURL
curl -X POST https://staging.rewardgenix.co.in/clientapi/generate-token \
  -H "x-cx-client-id: YOUR_CLIENT_ID" \
  -H "x-cx-client-secret: YOUR_CLIENT_SECRET"
JavaScript
const response = await fetch('https://staging.rewardgenix.co.in/clientapi/generate-token', {
  method: 'POST',
  headers: {
    'x-cx-client-id': 'YOUR_CLIENT_ID',
    'x-cx-client-secret': 'YOUR_CLIENT_SECRET'
  }
});

const { data } = await response.json();
const { accessToken, expiresIn, expiresAt } = data;
Node.js
const axios = require('axios');

const { data } = await axios.post(
  'https://staging.rewardgenix.co.in/clientapi/generate-token',
  {},
  {
    headers: {
      'x-cx-client-id': 'YOUR_CLIENT_ID',
      'x-cx-client-secret': 'YOUR_CLIENT_SECRET'
    }
  }
);

const token = data.data.accessToken;

Response — 200 OK

JSON
{
  "message": "Access token generated successfully",
  "data": {
    "accessToken": "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...",
    "expiresIn": 3600,
    "expiresAt": "2025-01-01T13:00:00.000Z",
    "tokenType": "Bearer"
  }
}

Error Responses

Status Code Description
401 ER_401_101 Invalid cxClientId or cxClientSecret
401 ER_401_102 Missing x-cx-client-id or x-cx-client-secret headers
429 ERR_429_001 Rate limit exceeded

Create Users (Batch) #

POST /clientapi/users

Create one or more end users in a single request. Duplicate userId or emailId entries are skipped with an error in the results array.

Headers

Header Required Description
Authorization Yes Bearer <accessToken>
x-cx-api-secret Yes Your API Secret
Content-Type Yes application/json

Required Fields

userId, fullName, firstName, lastName, emailId, phoneNumber

User Fields Reference

Field Type Required Description
userId string Yes External user identifier (unique)
fullName string Yes Full name of the user
firstName string Yes First name
lastName string Yes Last name
emailId string Yes Email address (unique)
phoneNumber string Yes Phone number
countryCode string No Country code (e.g. +91)
tier string No User tier (e.g. Gold, Silver)
branch string No Branch name
zone string No Zone
region string No Region
status string No ACTIVE, INACTIVE, or SUSPENDED
endUserType string No e.g. EMPLOYEE
catalogueType string No STANDARD or PREMIUM
flex1–flex10 string No Custom flexible fields
address object No Address with addressLine1/2, city, state, pincode

JSON
{
  "message": "Processed 2 users: 2 created, 0 failed",
  "data": {

Update Users (Batch) #

PUT /clientapi/users

Headers

Header Required Description
Authorization Yes Bearer <accessToken>
x-cx-api-secret Yes Your API Secret
Content-Type Yes application/json

Code Examples

cURL
curl -X PUT https://staging.rewardgenix.co.in/clientapi/users \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_ACCESS_TOKEN" \
  -H "x-cx-api-secret: YOUR_API_SECRET" \
  -d '{
    "users": [{
      "userId": "EMP001",
      "tier": "Platinum",
      "branch": "Corporate HQ"
    }]
  }'
JavaScript
const response = await fetch('https://staging.rewardgenix.co.in/clientapi/users', {

  headers: {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer YOUR_ACCESS_TOKEN',
    'x-cx-api-secret': 'YOUR_API_SECRET'
  },
  body: JSON.stringify({
    users: [{
      userId: 'EMP001',
      tier: 'Platinum',
      branch: 'Corporate HQ'
    }]
  })
});

const result = await response.json();
Node.js
const axios = require('axios');

const { data } = await axios.put(
  'https://staging.rewardgenix.co.in/clientapi/users',
  {
    users: [{
      userId: 'EMP001',
      tier: 'Platinum',
      branch: 'Corporate HQ'
    }]
  },
  {
    headers: {
      'Content-Type': 'application/json',
      'Authorization': 'Bearer YOUR_ACCESS_TOKEN',
      'x-cx-api-secret': 'YOUR_API_SECRET'

Response — 200 OK (Success)

JSON
{
  "message": "Processed 1 users: 1 updated, 0 failed",
  "data": {
    "results": [
      { "userId": "EMP001", "success": true }

  }

JSON

}

Get Users #

GET /clientapi/users

Retrieve end users with pagination and search. Search across name, email, phone, or userId.

Headers

Header Required Description
Authorization Yes Bearer <accessToken>
x-cx-api-secret Yes Your API Secret

Query Parameters

Param Type Default Description
page int 1 Page number
limit int 20 Results per page (max: 100)
search string — Search by name, email, phone, or userId
userId string — Filter by specific externalUserId

Code Examples

cURL
curl -X GET "https://staging.rewardgenix.co.in/clientapi/users?page=1&limit=20&search=john" \
  -H "Authorization: Bearer YOUR_ACCESS_TOKEN" \
  -H "x-cx-api-secret: YOUR_API_SECRET"
JavaScript
const params = new URLSearchParams({ page: 1, limit: 20, search: 'john' });

const response = await fetch(
  `https://staging.rewardgenix.co.in/clientapi/users?${params}`,
  {
    headers: {
      'Authorization': 'Bearer YOUR_ACCESS_TOKEN',
      'x-cx-api-secret': 'YOUR_API_SECRET'
    }
  }
);

const { data } = await response.json();
console.log(`Found ${data.pagination.total} users`);
Node.js
const axios = require('axios');

const { data } = await axios.get(
  'https://staging.rewardgenix.co.in/clientapi/users',
  {
    params: { page: 1, limit: 20, search: 'john' },
    headers: {
      'Authorization': 'Bearer YOUR_ACCESS_TOKEN',
      'x-cx-api-secret': 'YOUR_API_SECRET'
    }
  }
);

console.log(data.data.users);

Authentication Errors

Code HTTP Description Example
ER_401_101 401 Invalid or expired token {"message":"Invalid or expired token","status":"Error","code":"ER_401_101"}
ER_401_102 401 Missing or invalid client headers {"message":"Missing x-cx-client-id or x-cx-client-secret headers","status":"Error","code":"ER_401_102"}
ER_401_103 401 Invalid x-cx-api-secret {"message":"Invalid x-cx-api-secret","status":"Error","code":"ER_401_103"}
ER_403_001 403 IP not whitelisted {"message":"Access denied. IP not whitelisted.","status":"Error","code":"ER_403_001"}

Rate Limiting

Code HTTP Description Example
ERR_429_001 429 Rate limit exceeded {"message":"Rate limit exceeded. Please retry after 60 seconds.","status":"Error","code":"ERR_429_001"}

Validation Errors

Code HTTP Description Example
ER_400_001 400 Missing required fields {"message":"Validation failed: userId is required","status":"Error","code":"ER_400_001"}
ER_400_002 400 Invalid field format {"message":"Validation failed: emailId must be a valid email","status":"Error","code":"ER_400_002"}
ER_400_003 400 Empty users array {"message":"Validation failed: At least one user is required","status":"Error","code":"ER_400_003"}
ER_400_004 400 Unknown status value {"message":"Validation failed: status must be one of: ACTIVE, INACTIVE, SUSPENDED","status":"Error","code":"ER_400_004"}

Code HTTP Description Example
